  <script>
    const RADAR_CENTER = 160;
    const RADAR_RADIUS = 140;

    const polarToCart = (angle, radius) => {
      const rad = (angle * Math.PI) / 180 - Math.PI / 2;
      return {
        x: RADAR_CENTER + radius * Math.cos(rad),
        y: RADAR_CENTER + radius * Math.sin(rad)
      };
    };

    const generateBlips = () => {
      const blips = [];
      const victimCount = 1 + Math.floor(Math.random() * 2);
      const rescuerCount = 2 + Math.floor(Math.random() * 3);
      for (let i = 0; i < victimCount; i++) {
        blips.push({ type: 'victim', angle: Math.random() * 360, dist: 0.8 + Math.random() * 2.4, rssi: -72 - Math.floor(Math.random() * 20) });
      }
      for (let i = 0; i < rescuerCount; i++) {
        blips.push({ type: 'rescuer', angle: Math.random() * 360, dist: 0.5 + Math.random() * 2.0, rssi: -55 - Math.floor(Math.random() * 15) });
      }
      return blips;
    };

    const renderBlips = (blips) => {
      const g = document.getElementById('blips');
      g.innerHTML = '';
      let primaryVictim = null;
      blips.forEach(blip => {
        const radius = (blip.dist / 3.5) * RADAR_RADIUS;
        const pos = polarToCart(blip.angle, radius);
        const isVictim = blip.type === 'victim';
        if (isVictim) primaryVictim = blip;
        const color = isVictim ? '#FF3B30' : '#00FF41';
        const r = isVictim ? 6 : 4;
        const circle = document.createElementNS('http://www.w3.org/2000/svg', 'circle');
        circle.setAttribute('cx', pos.x);
        circle.setAttribute('cy', pos.y);
        circle.setAttribute('r', r);
        circle.setAttribute('fill', color);
        circle.setAttribute('stroke', '#F5F5F5');
        circle.setAttribute('stroke-width', '1');
        circle.setAttribute('stroke-opacity', '0.5');
        g.appendChild(circle);
      });
      const label = document.getElementById('floatingLabel');
      if (primaryVictim) {
        const pos = polarToCart(primaryVictim.angle, (primaryVictim.dist / 3.5) * RADAR_RADIUS);
        label.textContent = `DIST: ${primaryVictim.dist.toFixed(1)}m | RSSI: ${primaryVictim.rssi}dBm`;
        label.style.left = Math.max(8, Math.min(246, pos.x - 74)) + 'px';
        label.style.top = Math.max(8, Math.min(282, pos.y - 38)) + 'px';
        label.style.display = 'block';
      } else {
        label.style.display = 'none';
      }
    };

    let waveformData = Array(24).fill(0).map(() => 0.3 + Math.random() * 0.4);
    const waveformEl = document.getElementById('waveform');
    waveformEl.innerHTML = waveformData.map(() => '<div class="waveformBar"></div>').join('');
    const bars = waveformEl.querySelectorAll('.waveformBar');

    const updateWaveform = () => {
      waveformData = Array(24).fill(0).map(() => 0.2 + Math.random() * 0.7);
      const maxVal = Math.max(...waveformData);
      bars.forEach((bar, i) => {
        bar.style.height = Math.max(4, waveformData[i] * 48) + 'px';
        bar.classList.toggle('alert', waveformData[i] > 0.9);
      });
      const alertEl = document.getElementById('voiceAlert');
      if (maxVal > 0.92) {
        document.getElementById('voiceMatch').textContent = Math.min(98, Math.floor(90 + maxVal * 10));
        alertEl.style.display = 'block';
      } else {
        alertEl.style.display = 'none';
      }
    };

    let telemetry = { lat: 41.0082, long: 28.9784, airQuality: 'Good', deviceCount: 12 };
    const updateTelemetry = () => {
      telemetry.lat += (Math.random() - 0.5) * 0.0001;
      telemetry.long += (Math.random() - 0.5) * 0.0001;
      telemetry.airQuality = Math.random() > 0.85 ? 'Poor' : 'Good';
      telemetry.deviceCount = 10 + Math.floor(Math.random() * 6);
      document.getElementById('lat').textContent = telemetry.lat.toFixed(5) + 'Â°';
      document.getElementById('long').textContent = telemetry.long.toFixed(5) + 'Â°';
      const airEl = document.getElementById('airQuality');
      airEl.textContent = telemetry.airQuality;
      airEl.className = 'telemetryValue ' + (telemetry.airQuality === 'Good' ? 'good' : 'bad');
      document.getElementById('deviceCount').textContent = telemetry.deviceCount;
    };

    renderBlips(generateBlips());
    updateWaveform();
    bars.forEach((bar, i) => { bar.style.height = Math.max(4, waveformData[i] * 48) + 'px'; });

    setInterval(() => renderBlips(generateBlips()), 3000);
    setInterval(updateWaveform, 800);
    setInterval(updateTelemetry, 2000);
  </script>
